51 // Expand capacity safely.
52 void expand_capacity()
53 {
54 // double capacity until INT_MAX - 1 (- 1 for find())
55 capacity_ = (capacity_ < INT_MAX / 2) ? capacity_ * 2 : INT_MAX - 1;

57 // move data
58 T* tmp = new T[capacity_];

60 {
61 tmp[i] = data_[i];

63 delete[] data_;
64 data_ = tmp;

620 void sort(bool (*comparator)(const T& e1, const T& e2) = [](const T& e1, const T& e2)
621 { return e1 < e2; })
622 {
623 // for simplicity, bubble sort is usually enough
624

626 for (int i = 0; i < size_ - 1; i++)
627 {
628 swapped = false;
629 for (int j = 0; j < size_ - i - 1; j++)
630 {
631 if (comparator(data_[j + 1], data_[j]))
632 {
633 auto tmp = data_[j];
634 data_[j] = data_[j + 1];
635 data_[j + 1] = tmp;
636 swapped = true;
637 }
638 }
639 if (swapped == false)

680 List slice(int start, int stop, int step = 1) const
681 {
682 // although I have implemented the Python-like slice function (index can exceed the size),
683 // I finally decided to check the index for the simplicity of source code and the security of use, and consistent with the operator[]
684
685 // check
686 if (step == 0)
687 {
688 throw std::runtime_error("ERROR: Slice step cannot be zero.");
689 }
690
691 common::check_bounds(start, -size_, size_);
692 common::check_bounds(stop, -size_ - 1, size_ + 1);
693
694 // convert
695 start = start < 0 ? start + size_ : start;
696 stop = stop < 0 ? stop + size_ : stop;
697
698 // copy
699 List list;
700 for (int i = start; (step > 0) ? (i < stop) : (i > stop); i += step)
701 {
702 list += data_[i];
703 }
704
705 return list;
706 }
