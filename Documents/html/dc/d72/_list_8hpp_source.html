539 void insert(int index, const T& element)
540 {
541 // check
542 common::check_full(size_, MAX_CAPACITY);
543
544 common::check_bounds(index, -size_, size_ + 1);
545
546 // adjust capacity
547 if (size_ == capacity_)
548 {
549 // double capacity until MAX_CAPACITY
550 adjust_capacity((capacity_ < MAX_CAPACITY / 2) ? capacity_ * 2 : MAX_CAPACITY);
551 }
552
553 // index
554 index = index >= 0 ? index : index + size_;
555 for (int i = size_; i > index; --i)
556 {
557 data_[i] = data_[i - 1];
558 }
559
560 // insert
561 data_[index] = element; // copy assignment on T
562
563 // resize
564 ++size_;
565 }

802 List& adjust_capacity(int new_capacity)
803 {
804 if (new_capacity == 0)
805 {
806 throw std::runtime_error("ERROR: Capacity can not be zero.");
807 }
808
809 if (new_capacity < size_)
810 {
811 throw std::runtime_error("ERROR: Capacity can not be smaller than the size.");

847 List slice(int start, int stop, int step = 1) const
848 {
849 // although I have implemented the Python-like slice function (index can exceed the size),
850 // I finally decided to check the index for the simplicity of source code and the security of use, and consistent with the operator[]
851
852 // check
853 if (step == 0)
854 {
855 throw std::runtime_error("ERROR: Slice step can not be zero.");
856 }
857
858 common::check_bounds(start, -size_, size_);
859 common::check_bounds(stop, -size_ - 1, size_ + 1);
860
861 // convert
862 start = start < 0 ? start + size_ : start;
863 stop = stop < 0 ? stop + size_ : stop;
864
865 // copy
866 List list;
867 for (int i = start; (step > 0) ? (i < stop) : (i > stop); i += step)
868 {
869 list += data_[i];
870 }
871
872 return list;
