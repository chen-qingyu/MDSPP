787 List& adjust_capacity(int new_capacity = -1)
788 {
789 if (new_capacity == -1) // default
790 {
791 // double capacity until MAX_CAPACITY
792 capacity_ = (capacity_ < MAX_CAPACITY / 2) ? capacity_ * 2 : MAX_CAPACITY;
793 }
794 else // specified capacity
795 {
796 if (new_capacity == 0)
797 {
798 throw std::runtime_error("ERROR: Capacity can not be zero.");
799 }
800
801 if (new_capacity < size_)
802 {
803 throw std::runtime_error("ERROR: Capacity can not be smaller than the size.");
804 }
805
806 if (new_capacity > MAX_CAPACITY)
807 {
808 throw std::runtime_error("ERROR: Capacity can not be larger than the maximum capacity.");
809 }
810
811 capacity_ = new_capacity;
812 }
813
814 // move data
815 T* tmp = new T[capacity_];
816 for (int i = 0; i < size_; ++i)
817 {
818 tmp[i] = data_[i];
819 }
820 delete[] data_;
821 data_ = tmp;
822
823 return *this;
824 }

840 List slice(int start, int stop, int step = 1) const
841 {
842 // although I have implemented the Python-like slice function (index can exceed the size),
843 // I finally decided to check the index for the simplicity of source code and the security of use, and consistent with the operator[]
844
845 // check
846 if (step == 0)
847 {
848 throw std::runtime_error("ERROR: Slice step can not be zero.");
849 }
850
851 common::check_bounds(start, -size_, size_);
852 common::check_bounds(stop, -size_ - 1, size_ + 1);
853
854 // convert
855 start = start < 0 ? start + size_ : start;
856 stop = stop < 0 ? stop + size_ : stop;
857
858 // copy
859 List list;
860 for (int i = start; (step > 0) ? (i < stop) : (i > stop); i += step)
861 {
862 list += data_[i];
863 }
864
865 return list;
866 }
