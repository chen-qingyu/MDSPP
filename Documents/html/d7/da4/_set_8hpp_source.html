284 // Insert node.
285 Node* insert(Node* pos, const T& element)
286 {
287 // TODO: AVL
288 if (pos == nullptr)
289 {
290 pos = new Node(element);
291 size_++;
292 }
293 else
294 {
295 if (element < pos->data_)
296 {
297 pos->link_left(insert(pos->left_, element));
298 }
299 else if (pos->data_ < element)
300 {
301 pos->link_right(insert(pos->right_, element));

308 // Remove node.
309 Node* remove(Node* pos, const T& element)
310 {
311 // TODO: AVL
312 if (pos)
313 {
314 if (element < pos->data_)
315 {
316 pos->link_left(remove(pos->left_, element));
317 }
318 else if (pos->data_ < element)
319 {
320 pos->link_right(remove(pos->right_, element));
321 }
322 else // element == pos->data_
323 {
324 if (pos->left_ && pos->right_) // certainly not the min or max pos
325 {
326 Node* tmp = find_min(pos->right_);
327 pos->data_ = tmp->data_;
328 pos->link_right(remove(pos->right_, tmp->data_));
329 }
330 else // may be the min or max pos
331 {
332 // if it is the min or max pos, mark it
333 if (pos == min_)
334 {
335 min_ = end_;
336 }
337 if (pos == max_) // must `if`, not `else if` cuz pos may be root
338 {
339 max_ = end_;
340 }
341
342 Node* tmp = pos;
343 pos = pos->left_ ? pos->left_ : pos->right_;
344 delete tmp;
345 size_--;
346 }
347 }
348 }
349
350 return pos;

364 // Traverse the subtree rooted at the specified node.
365 template <typename F>
366 static void level_action(Node* node, F action)
367 {
368 // level order
369 if (node != nullptr)
370 {
371 Deque<Node*> queue;
372 queue.push_back(node);
373 while (!queue.is_empty())
374 {
375 node = queue.pop_front();
376 action(node->data_);
377 if (node->left_)
378 {
379 queue.push_back(node->left_);
380 }
381 if (node->right_)
382 {
383 queue.push_back(node->right_);
384 }
385 }
