35 int compare(const Integer& that) const
36 {
37 if (sign_ != that.sign_)
38 {
39 if (sign_ == '+') // this is +, that is - or 0
40 {
41 return 1; // gt
42 }
43 else if (sign_ == '-') // this is -, that is + or 0
44 {
45 return -1; // lt
46 }
47 else // this is 0, that is + or -
48 {

50 }
51 }
52
53 // the sign of two integers is the same
54
55 if (digits_.size_ != that.digits_.size_)
56 {
57 if (sign_ == '+')
58 {
59 return digits_.size_ > that.digits_.size_ ? 1 : -1;

62 {

64 }

68 {

518 Integer operator+(const Integer& rhs) const
519 {
520 // if one of the operands is zero, just return another one
521 if (sign_ == '0' || rhs.sign_ == '0')
522 {
523 return sign_ == '0' ? rhs : *this;
524 }
525
526 // if the operands are of opposite signs, perform subtraction
527 if (sign_ == '+' && rhs.sign_ == '-')
528 {
529 return *this - (-rhs);
530 }
531 else if (sign_ == '-' && rhs.sign_ == '+')
532 {
533 return rhs - (-*this);
534 }
535
536 // the sign of two integers is the same and not zero
537
538 // prepare variables
539 int size = std::max(digits_.size_, rhs.digits_.size_) + 1;
540
541 Integer num1 = *this;
542 num1.add_leading_zeros(size - 1 - num1.digits_.size_);
543
544 Integer num2 = rhs;
545 num2.add_leading_zeros(size - 1 - num2.digits_.size_);
546
547 Integer result;
548 result.sign_ = sign_; // the signs are same
549 result.add_leading_zeros(size - 1); // result initially has a 0
550
551 // simulate the vertical calculation
552 auto& a = num1.digits_;
553 auto& b = num2.digits_;
554 auto& c = result.digits_;
555 for (int i = 0; i < size - 1; i++)
556 {
557 c[i] += a[i] + b[i];
558 c[i + 1] = c[i] / 10;
559 c[i] %= 10;
560 }
561
562 // remove leading zeros and return result
563 return result.remove_leading_zeros();
564 }

572 Integer operator-(const Integer& rhs) const
573 {
574 // if one of the operands is zero
575 if (sign_ == '0' || rhs.sign_ == '0')
576 {
577 return sign_ == '0' ? -rhs : *this;
578 }
579
580 // if the operands are of opposite signs, perform addition
581 if (sign_ == '+' && rhs.sign_ == '-')
582 {
583 return *this + (-rhs);
584 }
585 else if (sign_ == '-' && rhs.sign_ == '+')
586 {
587 return *this + (-rhs);
588 }

590 // the sign of two integers is the same and not zero
591
592 // prepare variables
593 int size = std::max(digits_.size_, rhs.digits_.size_);
594
595 Integer num1 = *this;
596 num1.add_leading_zeros(size - num1.digits_.size_);
597
598 Integer num2 = rhs;
599 num2.add_leading_zeros(size - num2.digits_.size_);
600
601 Integer result;
602 result.sign_ = sign_; // the signs are same
603 if (sign_ == '+' ? num1 < num2 : num1 > num2) // let num1.abs() >= num2.abs()
604 {
605 common::swap(num1, num2);
606 result = -result;
607 }
608 result.add_leading_zeros(size - 1); // result initially has a 0
609
610 // simulate the vertical calculation, assert a >= b
611 auto& a = num1.digits_;
612 auto& b = num2.digits_;
613 auto& c = result.digits_;
614 for (int i = 0; i < size; i++)
615 {
616 if (a[i] < b[i]) // carry
617 {
618 a[i + 1]--;
619 a[i] += 10;
620 }
621 c[i] = a[i] - b[i];
622 }
623
624 // remove leading zeros
625 result.remove_leading_zeros();
626
627 // if result is zero, set sign to '0'
628 result.sign_ = (result.digits_.size_ == 1 && result.digits_[0] == 0) ? '0' : result.sign_;
629

631 return result;

640 Integer operator*(const Integer& rhs) const
641 {
642 // if one of the operands is zero, just return zero
643 if (sign_ == '0' || rhs.sign_ == '0')

681 Integer operator/(const Integer& rhs) const
682 {
683 // if rhs is zero, throw an exception
684 if (rhs.sign_ == '0')
685 {
686 throw std::runtime_error("ERROR: Divide by zero.");
687 }
688
689 // if this is zero, just return zero
690 if (sign_ == '0')
691 {
692 return 0;
693 }
694
695 // the sign of two integers is not zero
696
697 // prepare variables
698 int size = digits_.size_ - rhs.digits_.size_ + 1;
699
700 Integer num1 = (*this).abs();
701
702 Integer tmp; // intermediate variable for rhs * 10^i
703 tmp.sign_ = '+'; // positive
704
705 Integer result;
706 result.sign_ = (sign_ == rhs.sign_ ? '+' : '-'); // the sign is depends on the sign of operands
707 result.add_leading_zeros(size - 1); // result initially has a 0
708
709 // calculation
710 const auto& b = rhs.digits_;
711 auto& c = result.digits_;
712 for (int i = size - 1; i >= 0; i--)
713 {
714 tmp.digits_ = List<signed char>({0}) * i + b; // tmp = rhs * 10^i in O(N)
715
716 while (num1 >= tmp) // <= 9 loops, so O(1)
717 {
718 c[i]++;
719 num1 -= tmp;
720 }
721 }
722
723 // if result is zero, set sign to '0'
724 result.sign_ = (result.digits_.size_ == 1 && result.digits_[0] == 0) ? '0' : result.sign_;
725
726 // remove leading zeros and return
727 return result.remove_leading_zeros();
728 }

748 Integer pow(const Integer& n, const Integer& mod = 0) const
749 {
750 if (n.is_negative())
751 {
752 return 0;
753 }
754
755 if (mod.is_zero())
756 {
757 if (n.is_zero())
758 {
759 return 1;
760 }
761 else if (n.is_even())
762 {
763 Integer y = pow(n / 2);
764 return y * y;
765 }
766 else // n is odd
767 {
768 Integer y = pow(n / 2); // integer divide
769 return *this * y * y;
770 }
771 }
772 else // fast power algorithm
773 {
774 Integer num = *this;
775 Integer exp = n;
776 Integer result = 1;
777
778 num %= mod;
779
780 while (!exp.is_zero())
781 {
782 if (exp.is_odd())
783 {
784 result = (result * num) % mod;
785 }
786 num = (num * num) % mod;
787 exp /= 2;
788 }
789 return result;
790 }
