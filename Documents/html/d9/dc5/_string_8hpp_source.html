92 // Use the KMP algorithm to find the position of the pattern.
93 static int kmp(const char* this_str, const char* patt_str, int n, int m)
94 {
95 if (n < m)
96 {
97 return -1;
98 }
99
100 if (m == 0) // "" is in any string at index 0.
101 {
102 return 0;
103 }
104
105 int* match = new int[m];
106 match[0] = -1;
107
108 for (int j = 1; j < m; j++)
109 {
110 int i = match[j - 1];
111 while ((i >= 0) && (patt_str[i + 1] != patt_str[j]))
112 {
113 i = match[i];
114 }
115 match[j] = (patt_str[i + 1] == patt_str[j]) ? i + 1 : -1;
116 }
117
118 int s = 0;
119 int p = 0;
120 while (s < n && p < m)
121 {
122 if (this_str[s] == patt_str[p])
123 {
124 s++;
125 p++;
126 }
127 else if (p > 0)
128 {
129 p = match[p - 1] + 1;
130 }
131 else
132 {
133 s++;
134 }
135 }
136
137 delete[] match;
138
139 return (p == m) ? (s - m) : -1;
140 }

514 double to_decimal() const
515 {
516 // check infinity or nan
517 static const char* pos_infs[12] = {"inf", "INF", "Inf", "+inf", "+INF", "+Inf", "infinity", "INFINITY", "Infinity", "+infinity", "+INFINITY", "+Infinity"};
518 static const char* neg_infs[6] = {"-inf", "-INF", "-Inf", "-infinity", "-INFINITY", "-Infinity"};
519 static const char* nans[9] = {"nan", "NaN", "NAN", "+nan", "+NaN", "+NAN", "-nan", "-NaN", "-NAN"};
520
521 for (int i = 0; i < 12; ++i)
522 {
523 if (*this == pos_infs[i])
524 {
525 return INFINITY;
526 }
527 }
528 for (int i = 0; i < 6; ++i)
529 {
530 if (*this == neg_infs[i])
531 {
532 return -INFINITY;

535 for (int i = 0; i < 9; ++i)

545 double sign = 1; // default '+'
546 double decimal_part = 0;
547 int decimal_cnt = 0;
548 double exp_sign = 1; // default '+'
549 int exp_part = 0;
550
551 // FSM
552 state st = S_BEGIN_BLANK;
553 for (int i = 0; i < list_.size_; ++i)
554 {
555 event ev = get_event(list_.data_[i], 10);
556 switch (st | ev)
557 {
558 case S_BEGIN_BLANK | E_BLANK:
559 st = S_BEGIN_BLANK;
560 break;
561
562 case S_BEGIN_BLANK | E_SIGN:
563 sign = (list_.data_[i] == '+') ? 1 : -1;
564 st = S_SIGN;
565 break;
566
567 case S_BEGIN_BLANK | E_DEC_POINT:
568 case S_SIGN | E_DEC_POINT:
569 st = S_DEC_POINT_NOT_LEFT;
570 break;
571
572 case S_BEGIN_BLANK | E_NUMBER:
573 case S_SIGN | E_NUMBER:
574 case S_INT_PART | E_NUMBER:
575 decimal_part = decimal_part * 10 + char_to_integer(list_.data_[i], 10);
576 st = S_INT_PART;
577 break;
578
579 case S_INT_PART | E_DEC_POINT:
580 st = S_DEC_POINT_HAS_LEFT;
581 break;
582
583 case S_DEC_POINT_NOT_LEFT | E_NUMBER:
584 case S_DEC_PART | E_NUMBER:
585 case S_DEC_POINT_HAS_LEFT | E_NUMBER:
586 decimal_part = decimal_part * 10 + char_to_integer(list_.data_[i], 10);
587 decimal_cnt++;
588 st = S_DEC_PART;
589 break;
590
591 case S_INT_PART | E_EXP:
592 case S_DEC_POINT_HAS_LEFT | E_EXP:
593 case S_DEC_PART | E_EXP:
594 st = S_EXP;
595 break;
596
597 case S_EXP | E_SIGN:
598 exp_sign = (list_.data_[i] == '+') ? 1 : -1;
599 st = S_EXP_SIGN;
600 break;
601
602 case S_EXP | E_NUMBER:
603 case S_EXP_SIGN | E_NUMBER:
604 case S_EXP_PART | E_NUMBER:
605 exp_part = exp_part * 10 + char_to_integer(list_.data_[i], 10);
606 st = S_EXP_PART;
607 break;
608
609 case S_INT_PART | E_BLANK:
610 case S_DEC_POINT_HAS_LEFT | E_BLANK:
611 case S_DEC_PART | E_BLANK:
612 case S_EXP_PART | E_BLANK:
613 case S_END_BLANK | E_BLANK:
614 st = S_END_BLANK;
615 break;
616
617 default:
618 st = S_OTHER;
619 i = list_.size_; // exit for loop
620 break;
621 }
622 }
623 if (st != S_INT_PART && st != S_DEC_POINT_HAS_LEFT && st != S_DEC_PART && st != S_EXP_PART && st != S_END_BLANK)
624 {
625 throw std::runtime_error("ERROR: Invalid literal for to_decimal().");
626 }
627
628 return sign * ((decimal_part / std::pow(10, decimal_cnt)) * std::pow(10, exp_sign * exp_part));
