55 // Expand capacity safely.
56 void expand_capacity()
57 {
58 // double capacity until MAX_CAPACITY
59 capacity_ = (capacity_ < MAX_CAPACITY / 2) ? capacity_ * 2 : MAX_CAPACITY;
60
61 // move data
62 T* tmp = new T[capacity_];
63 for (int i = 0; i < size_; ++i)
64 {
65 tmp[i] = data_[i];
66 }
67 delete[] data_;
68 data_ = tmp;
69 }

422 void insert(int index, const T& element)
423 {
424 // check
425 common::check_full(size_, MAX_CAPACITY);
426
427 common::check_bounds(index, -size_, size_ + 1);
428
429 // expand capacity
430 if (size_ == capacity_)
431 {
432 expand_capacity();
433 }
434
435 // index
436 index = index >= 0 ? index : index + size_;
437 for (int i = size_; i > index; --i)
438 {
439 data_[i] = data_[i - 1];
440 }
441
442 // insert

444
445 // resize
446 ++size_;
447 }

641 List& sort(bool (*comparator)(const T& e1, const T& e2) = [](const T& e1, const T& e2)
642 { return e1 < e2; })
643 {
644 // for simplicity, bubble sort is usually enough
645
646 for (int i = 0; i < size_ - 1; i++)
647 {
648 bool swapped = false;
649 for (int j = 0; j < size_ - i - 1; j++)
650 {
651 if (comparator(data_[j + 1], data_[j]))
652 {
653 common::swap(data_[j], data_[j + 1]);
654 swapped = true;
655 }
656 }
657 if (swapped == false)
658 {
659 break;
660 }
661 }
662
663 return *this;
664 }

692 List slice(int start, int stop, int step = 1) const
693 {
694 // although I have implemented the Python-like slice function (index can exceed the size),
695 // I finally decided to check the index for the simplicity of source code and the security of use, and consistent with the operator[]
696
697 // check
698 if (step == 0)
699 {
700 throw std::runtime_error("ERROR: Slice step cannot be zero.");
701 }
702
703 common::check_bounds(start, -size_, size_);
704 common::check_bounds(stop, -size_ - 1, size_ + 1);
705
706 // convert
707 start = start < 0 ? start + size_ : start;
708 stop = stop < 0 ? stop + size_ : stop;
709
710 // copy
711 List list;
712 for (int i = start; (step > 0) ? (i < stop) : (i > stop); i += step)
713 {
714 list += data_[i];
715 }
716
717 return list;
718 }
